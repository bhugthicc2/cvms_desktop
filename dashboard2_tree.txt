|   |       emit(
|   |         state.copyWith(

|   |         ),
|   |       );
|   |     }
|   |   
|   |     void backToPreviousView() {
|   |       final previousView = state.previousViewMode;
|   |       if (previousView != null) {
|   |         emit(state.copyWith(viewMode: previousView));

|   |         // Fallback to global if no previous view

|   |       }
|   |     }

|   |       showGlobalDashboard();
|   |     }

|   |       emit(state.copyWith(error: null));
